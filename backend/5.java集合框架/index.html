1.基础知识

逻辑上表现为：线性表，图，树，堆，栈

物理上表现为：紧密顺序结构，跳转链式结构

集合用于存储引用类型数据，具体分为：

    • List子接口

2.Collection接口

package wiki.mrx.www;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Iterator;

public class TestCollection {
    public static void main(String[] args) {
        //创建对象
        Collection col = new ArrayList();

        //增
        col.add(18);
        col.add(18);
        col.add(18);
        col.add(18);

        //增强型for循环遍历
        for (Object o: col) {
            System.out.println(o);
        }
        
        //iterator遍历
        Iterator it = col.iterator();
        while (it.hasNext()){
            System.out.println(it.next());
        }

        //删
        col.clear();
        col.remove(18);
        
        //查
        col.size();
        col.isEmpty();
        col.equals(null);
        col.contains(18);
        
        System.out.println(col);
    }
}

2.1 List接口

package wiki.mrx.www;

import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;

public class TestList {
    public static void main(String[] args) {
        //创建对象
        List list = new ArrayList();

        //增加
        list.add(18);
        list.add(18);
        list.add(18);
        list.add(3,88);

        //删除
        list.remove(2);

        //修改
        list.set(2,99);

        //查询
        for (int i = 0; i < list.size(); i++) {
            System.out.println(list.get(i));
        }

        for (Object o : list){
            System.out.println(o);
        }

        Iterator it = list.iterator();
        while (it.hasNext()){
            System.out.println(it.next());
        }
    }
}

2.1.2 LinkedList实现类jdk1.8源码

public class LinkedList<E>//源码底层泛型为E，说明数据类型在创建对象是确定
//存储空间是包含头尾节点的链表node
transient Node<E> first;
transient Node<E> last;
Node(Node<E> prev, E element, Node<E> next) {
    this.item = element;
    this.next = next;
    this.prev = prev;
 }
//add添加时
void linkLast(E e) {
    //last节点给1，如果是第一个元素1为null
    final Node<E> l = last;
    //包装对象
    final Node<E> newNode = new Node<>(l, e, null);
    //last节点指向包装好的对象
    last = newNode;
    if (l == null)//如果添加的是第一个节点
        first = newNode;//first指向新节点
    else//如果添加的不是第一个节点
        l.next = newNode;//将l的下一个指向新节点
    size++;//集合元素数量加1
    modCount++;
}
