1.泛型

泛型：检测或设置引用数据类型

  • 在集合中加泛型可以规定存入集合的数据类型，便于后续遍历
  • 在类中加泛型可以后期传入具体的数据类型，而不必提前规定
package wiki.mrx.www;

    public void test(E m){

public class Test {
    public static void main(String[] args) {
       Testa<String> t = new Testa<>();
       t.test("test");
    }
}

package wiki.mrx;

        return seasonName;
    }

    public String getSeasonDesc() {
        return seasonDesc;
    }

    @Override
    public String toString() {
        return "Season{" +
                "seasonName='" + seasonName + '\'' +
                ", seasonDesc='" + seasonDesc + '\'' +
                '}';
    }
}

5.反射reflect

反射：动态获取和调用类或对象信息的机制

package wiki.mrx;

import java.util.Arrays;

public class TestReflect {
    public static void main(String[] args) throws Exception {
        Person person = new Person();
        System.out.println(person.getClass());
        System.out.println(Person.class);
        System.out.println(Class.forName("wiki.mrx.Person"));//调用class静态方法
        System.out.println(TestReflect.class.getClassLoader().loadClass("wiki.mrx.Person"));
        System.out.println(Person.class.getConstructor());//获得构造器
        System.out.println(Person.class.newInstance());//创造对象
        System.out.println(Arrays.toString(Person.class.getFields()));//获取属性
        System.out.println(Arrays.toString(Person.class.getMethods()));//获取方法
    }
}

//awt例子
package wiki.mrx;

import java.awt.*;
import java.awt.event.WindowAdapter;
import java.awt.event.WindowEvent;

public class TestGUI {
    public static void main(String[] args) {
        //1.新建窗口和面板
        Frame frame = new Frame();
        Panel panel = new Panel();

        //2.窗口属性设置
        frame.setLayout(null);
        frame.setBounds(300,300,500,500);
        frame.setBackground(new Color(3,6,9));

        //3.面板属性设置
        panel.setBounds(50,50,400,400);
        panel.setBackground(new Color(0, 251, 141));

        //4.窗口中放置面板
        frame.add(panel);
        frame.setVisible(true);

        //5.创建监听事件
        frame.addWindowListener(new WindowAdapter() {
            @Override
            public void windowClosing(WindowEvent e) {
                System.exit(0);
            }
        });
    }
}
